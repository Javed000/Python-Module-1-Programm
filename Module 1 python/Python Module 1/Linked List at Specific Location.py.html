class node():
    def __init__(self,data):
        self.data = data
        self.next = None

class Linkedlist():

    def __init__(self):
        self.head = None

    def insert_at_specific(self, pos, data):
        np = node(data)
        temp = self.head
        for i in range(pos-1):
            temp = temp.next
        np.data = data
        np.next = temp.next
        temp.next = np
    def display(self):
        if self.head == None:
            print("Linked List Is EMPTY")


        temp = self.head
        while temp:
            print(temp.data,"-->", end = "")
            temp = temp.next

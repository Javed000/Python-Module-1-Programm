class student:
    def __init__(self,m1,m2):
        self.m1 = m1
        self.m2 = m2

    def __ge__(self, other):
       s1 = self.m1 + other.m2
       s2 =self.m2 +  other.m2
       if s1<=s2:
           return True
       else:
            False

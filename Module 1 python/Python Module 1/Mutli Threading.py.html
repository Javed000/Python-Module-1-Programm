# Multi Threading --> Threading is used to call a no. of function simultaneoulsy in order run a output randomly or a parallely is called Threading.

from threading import *    # Firstly it Required to Import Thread.
class student(Thread):
    def number(self):
        for i in range (500):
            print('LOVE JAVED')

class stud(Thread):
    def number(self):
        for i in range (500):
            print('HEY')


s1 = student()
s2 = stud()

s1.start()     # We use start as a function name to call a thread.
s2.start()

class node():
    def __init__(self,data):
        self.data = data
        self.next = None

class Linkedlist():

    def __init__(self):
        self.head = None

    def insert_at_begining(self,data):
        nb = node(data)
        nb.next = self.head
        self.head = nb

    def display(self):
        if self.head == None:
            print("Linked List Is EMPTY")


        temp = self.head
        while temp:
            print(temp.data,"-->", end = "")

            temp = temp.next

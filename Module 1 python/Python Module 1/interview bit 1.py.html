class Solution:
    # @param A : list of integers
    # @param B : list of integers
    # @return an integer
    def coverPoints(self, A, B):
        sum=0
        n=len(A)
        for i in range(1,n):
            a=(A[i]-A[i-1])
            a=abs(a)
            b=B[i]-B[i-1]
            b=abs(b)
            sum=sum+max(a,b)
        return sum

c1 = Solution()
A = [0, 1, 1]
B = [0, 1, 2]
print(c1.coverPoints(A,B))

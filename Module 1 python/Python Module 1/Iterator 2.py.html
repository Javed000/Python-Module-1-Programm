class student():
    def __init__(self):
        self.num = 1

    def  __iter__(self):
        return self

    def __next__(self):

       if self.num <= 50:
        val = self.num
        self.num += 1
        return val
       else:
        raise StopIteration



s1 = student()

print(s1.__next__())    # Print the number next to the number 1.
print(s1.__next__())
print(s1.__next__())
print(s1.__next__())

for i in s1:            # Create a infinite number as a output starting with a number 1.
    print(i)

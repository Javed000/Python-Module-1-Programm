Double Linked List At Specific Location.py 
class node():
    def __init__(self,data):
        self.data = data
        self.next = None
        self.prev = None


class dll():
    def __init__(self):
        self.head = None

    def insert_at_specific(self,pos,data):
        ns = node(data)
        temp = self.head
        for i in range(1,pos-1):
            temp = temp.next
        ns.prev = temp
        ns.next = temp.next
        temp.next = ns

        temp.next.prev = ns



    def display(self):
        if self.head == None:
            print("Empty Double Linked List")

        temp = self.head
        while temp:
            print(temp.data,"-->",end="")
            temp = temp.next




n1 = node(100)
l = dll()
l.head = n1
n2 = node(200)
n1.next = n2
n2.prev = n1
n3 = node(300)
n2.next = n3
n3.prev = n2
l.insert_at_specific(3,50)
l.display()

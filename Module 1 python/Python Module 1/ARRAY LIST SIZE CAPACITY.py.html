ARRAY LIST SIZE CAPACITY.py 
import sys
def arraysize(list):
    print('Capacity:', sys.getsizeof(list))

    print('Size:', len(list))

    print("Space Left :", sys.getsizeof(list) - len(list))


list = []
print('Empty list is created')
print('Details of a list')
arraysize(list)

print()
list.append('Sugar')
print('After Adding Sugar In A List')
print('The Required List Is')
print(list)
arraysize(list)


# If we want to add a new items in a list:

list.insert(0,'MILK')
print('After insertion the required list is')
print(list)
arraysize(list)


print()
list.pop(1)
print('After Deletion, The required list is')
print(list)
arraysize(list)

class node():
    def __init__(self,data):
        self.data = data
        self.next = None

class sll():
    def __init__(self): 
        self.head = None
        
        
    def display(self):
        temp = self.head
        if self.head is None:
            print("Empty")
            
        while temp:
            print(temp.data,"-->",end=" ")
            print(temp.next)
            temp = temp.next
            
        
L = sll()

def search(list,n):

    l = 0
    u = len(list)-1
    while l<=u:
        mid =(l + u) //2
        if list[mid]== n:

            return  True

        else :
            if list[mid]<n:

                l = mid+1
            else:
                n = mid+1

    return False


list=[1,3,5,7,9,10]
n = 1
if search(list,n):
    print('Required Number Is Found')

else:
    print('Required Number Is Not Found')

class employee():
    increment = 3
    def __init__(self, fname, lname, salary):
        self.fname = fname
        self.lname = lname
        self.salary = salary
        self.increment = 1

    def increase(self):
        self.salary =  self.salary * self.increment

e1 = employee('Javed', 'Akthar', '40505')
print(e1.salary)
e1.increase()

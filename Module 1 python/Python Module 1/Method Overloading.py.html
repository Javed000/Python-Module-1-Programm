class student:
    def __init__(self,a,b):
        self.a = a
        self.b = b

    def sum(self,c,d):

        c = c +d
        return c

    def sum2(self,e,f,g,h):

        l =e + f + g + h
        return l



s1 =student(22,44)
print(s1.sum(5,7))
